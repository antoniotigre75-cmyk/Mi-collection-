    function setLanguage(lang) {
        const enContent = document.getElementById('en-content');
        const esContent = document.getElementById('es-content');
        const buttons = document.querySelectorAll('.lang-btn');

        buttons.forEach(btn => btn.classList.remove('active'));

        if (lang === 'en') {
            enContent.classList.remove('hidden');
            esContent.classList.add('hidden');
            buttons[0].classList.add('active');
        } else {
            enContent.classList.add('hidden');
            esContent.classList.remove('hidden');
            buttons[1].classList.add('active');
        }
    }

    function addPhotos(event) {
        const files = event.target.files;
        const galleries = [
            document.getElementById('gallery-en'),
            document.getElementById('gallery-es')
        ];

        Array.from(files).forEach(file => {
            const reader = new FileReader();
            reader.onload = function(e) {
                const galleryItem = document.createElement('div');
                galleryItem.className = 'gallery-item';
                galleryItem.innerHTML = `
                    <img src="${e.target.result}" alt="Added photo">
                    <div class="gallery-label">New Addition</div>
                    <button class="delete-btn" onclick="deletePhoto(this)">×</button>
                `;

                galleries.forEach(gallery => {
                    gallery.appendChild(galleryItem.cloneNode(true));
                    gallery.lastChild.querySelector('.delete-btn').onclick = function() {
                        deletePhoto(this);
                    };
                });
            };
            reader.readAsDataURL(file);
        });

        event.target.value = '';
    }

    function deletePhoto(button) {
        const item = button.parentElement;
        const gallery = item.parentElement;
        const index = Array.from(gallery.children).indexOf(item);

        const galleries = [
            document.getElementById('gallery-en'),
            document.getElementById('gallery-es')
        ];

        galleries.forEach(gallery => {
            if (gallery.children[index]) {
                gallery.children[index].remove();
            }
        });
    }
</script>
